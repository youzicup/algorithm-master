class Solution {
public:
  int lengthOfLongestSubstring(string s) {
    unordered_map<char, int> m;
    int maxLen = 0, left = 0;
    for(int i = 0; i < s.size(); i++) {
      if(m.find(s[i]) != m.end() && m[s[i]] >= left) {
        left = m[s[i]] + 1;
      } else {
        maxLen = max(i - left + 1, maxLen);
      }
      m[s[i]] = i;
    }
    return maxLen;
  }
};
