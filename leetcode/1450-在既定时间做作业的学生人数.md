class Solution {
public:
  int busyStudent(vector<int>& startTime, vector<int>& endTime, int queryTime) {
    int count = 0;
    for(int t : startTime) {
      if(t <= queryTime) count++;
    }
    for(int t : endTime) {
      if(t < queryTime) count--;
    }
    return count;
  }
};
