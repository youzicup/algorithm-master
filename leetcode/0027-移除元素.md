**JavaScript:**
```javascript
/**
 * @param {number[]} nums
 * @param {number} val
 * @return {number}
 */
var removeElement = function(nums, val) {
  var len = nums.length
  var temp = null
  for (i=0;i<len;i++) {
    if(nums[i] == val) {
      temp = nums[i]
      nums[i--] = nums[len-1]
      nums[len-1] = temp
      if(i>=--len) break
    }
  }
  return len
};
```

**C++:**
```C++
class Solution {
public:
  int removeElement(vector<int>& nums, int val) {
    int len = nums.size();
    for (int i = 0; i < len; i++) {
      if(nums[i] == val) {
        nums[i] ^= nums[--len];
        nums[len] ^= nums[i];
        nums[i--] ^= nums[len];
        if(i >= len) break;
      }
    }
    return len;
  }
};
```
